#version 400 core

in vec3 lightVector;
in vec3 surfaceNormal;
in vec2 texCoords;

uniform sampler2D texture_diffuse;

out vec4 out_color;

uniform vec3 lightColor;

void main(void){
	float val = dot(lightVector, surfaceNormal);
	float brightness = max(val, 0.3);
	vec3 color = brightness * lightColor;
	
    out_color = texture(texture_diffuse, texCoords) * vec4(color, 1.0);
}
